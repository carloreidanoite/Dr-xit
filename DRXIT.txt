local ammoConnections = {}
local function toggleInfiniteAmmo(state)
    ammoInfiniteEnabled = state
    
    for _, connection in pairs(ammoConnections) do
        if connection then connection:Disconnect() end
    end
    ammoConnections = {}
    
    if state then
        local function applyInfiniteAmmo(weapon)
            if not weapon or not weapon:IsA("Tool") then return end
            
            local ammo = weapon:FindFirstChild("CurrentAmmo")
            if ammo and ammo:IsA("NumberValue") then
                ammo.Value = 42
                
                local connection = ammo.Changed:Connect(function()
                    if ammoInfiniteEnabled and ammo.Value ~= 42 then
                        ammo.Value = 42
                    end
                end)
                table.insert(ammoConnections, connection)
                
                local watchdogConnection
                watchdogConnection = RunService.Heartbeat:Connect(function()
                    if not ammoInfiniteEnabled then
                        watchdogConnection:Disconnect()
                        return
                    end
                    if ammo.Parent and ammo.Value ~= 42 then
                        ammo.Value = 42
                    end
                end)
                table.insert(ammoConnections, watchdogConnection)
            end
        end
        
        local function monitorWorkspace()
            for _, obj in pairs(workspace:GetChildren()) do
                if obj.Name == "UMP45" and obj:IsA("Tool") then
                    applyInfiniteAmmo(obj)
                end
            end
            
            local workspaceConnection = workspace.ChildAdded:Connect(function(child)
                if ammoInfiniteEnabled and child.Name == "UMP45" and child:IsA("Tool") then
                    wait(0.1)
                    applyInfiniteAmmo(child)
                end
            end)
            table.insert(ammoConnections, workspaceConnection)
        end
        
        local function monitorBackpack()
            local backpack = plr:FindFirstChild("Backpack")
            if backpack then
                for _, obj in pairs(backpack:GetChildren()) do
                    if obj.Name == "UMP45" and obj:IsA("Tool") then
                        applyInfiniteAmmo(obj)
                    end
                end
                
                local backpackConnection = backpack.ChildAdded:Connect(function(child)
                    if ammoInfiniteEnabled and child.Name == "UMP45" and child:IsA("Tool") then
                        wait(0.1)
                        applyInfiniteAmmo(child)
                    end
                end)
                table.insert(ammoConnections, backpackConnection)
            end
        end
        
        local function onCharacterAdded(character)
            if not ammoInfiniteEnabled then return end
            
            local backpack = plr:WaitForChild("Backpack", 10)
            if backpack then
                monitorBackpack()
            end
            
            local toolConnection = character.ChildAdded:Connect(function(child)
                if ammoInfiniteEnabled and child.Name == "UMP45" and child:IsA("Tool") then
                    wait(0.1)
                    applyInfiniteAmmo(child)
                end
            end)
            table.insert(ammoConnections, toolConnection)
        end
        
        if plr.Character then
            onCharacterAdded(plr.Character)
        end
        
        local characterConnection = plr.CharacterAdded:Connect(onCharacterAdded)
        table.insert(ammoConnections, characterConnection)
        
        monitorWorkspace()
        
        spawn(function()
            while ammoInfiniteEnabled do
                for _, obj in pairs(workspace:GetChildren()) do
                    if obj.Name == "UMP45" and obj:IsA("Tool") then
                        local ammo = obj:FindFirstChild("CurrentAmmo")
                        if ammo and ammo:IsA("NumberValue") and ammo.Value ~= 42 then
                            ammo.Value = 42
                        end
                    end
                end
                
                if plr:FindFirstChild("Backpack") then
                    for _, obj in pairs(plr.Backpack:GetChildren()) do
                        if obj.Name == "UMP45" and obj:IsA("Tool") then
                            local ammo = obj:FindFirstChild("CurrentAmmo")
                            if ammo and ammo:IsA("NumberValue") and ammo.Value ~= 42 then
                                ammo.Value = 42
                            end
                        end
                    end
                end
                
                if plr.Character then
                    local equippedWeapon = plr.Character:FindFirstChild("UMP45")
                    if equippedWeapon and equippedWeapon:IsA("Tool") then
                        local ammo = equippedWeapon:FindFirstChild("CurrentAmmo")
                        if ammo and ammo:IsA("NumberValue") and ammo.Value ~= 42 then
                            ammo.Value = 42
                        end
                    end
                end
                
                wait(0.5)
            end
        end)
    end
end

local function toggleNoRecoil(state)
    noRecoilEnabled = state
    if state then
        local function protectForPlayer(player)
            local backpack = player:WaitForChild("Backpack", 10)
            local UMP45 = backpack:FindFirstChild("UMP45") or backpack:WaitForChild("UMP45", 15)
            if not UMP45 then return end

            local config = UMP45:FindFirstChild("Configuration") or UMP45:WaitForChild("Configuration", 10)
            if not config then return end

            local obj = config:FindFirstChild("TotalRecoilMax") or config:WaitForChild("TotalRecoilMax", 5)
            if not obj then return end

            if obj:IsA("NumberValue") then
                obj.Value = 0
                obj.Changed:Connect(function()
                    if noRecoilEnabled and obj.Value ~= 0 then
                        obj.Value = 0
                    end
                end)

                task.spawn(function()
                    while noRecoilEnabled and player.Parent do
                        if obj.Value ~= 0 then
                            obj.Value = 0
                        end
                        task.wait(0.2)
                    end
                end)
            end
        end
        spawn(function() protectForPlayer(plr) end)
    end
end

local function applyShotCooldown(value)
    shotCooldownValue = value
    local function protectForPlayer(player)
        local backpack = player:WaitForChild("Backpack", 10)
        local UMP45 = backpack:FindFirstChild("UMP45") or backpack:WaitForChild("UMP45", 15)
        if not UMP45 then return end

        local config = UMP45:FindFirstChild("Configuration") or UMP45:WaitForChild("Configuration", 10)
        if not config then return end

        local obj = config:FindFirstChild("ShotCooldown") or config:WaitForChild("ShotCooldown", 5)
        if not obj then return end

        if obj:IsA("NumberValue") then
            obj.Value = value
            obj.Changed:Connect(function()
                if obj.Value ~= value then
                    obj.Value = value
                end
            end)

            task.spawn(function()
                while player.Parent do
                    if obj.Value ~= value then
                        obj.Value = value
                    end
                    task.wait(0.2)
                end
            end)
        end
    end
    spawn(function() protectForPlayer(plr) end)
end

local function createESP(player)
    if player == plr or espObjects[player] then return end
    
    local quad = Drawing.new("Quad")
    local line = Drawing.new("Line")
    local name = Drawing.new("Text")
    local health = Drawing.new("Text")
    local distance = Drawing.new("Text")

    quad.Thickness = 2
    quad.Filled = false
    quad.Visible = false
    
    line.Thickness = 1.5
    line.Visible = false
    
    name.Size = 14
    name.Outline = true
    name.Center = true
    name.Visible = false
    
    health.Size = 13
    health.Outline = true
    health.Center = true
    health.Visible = false
    
    distance.Size = 13
    distance.Outline = true
    distance.Center = true
    distance.Visible = false

    espObjects[player] = {
        Box = quad,
        Line = line,
        Name = name,
        Health = health,
        Distance = distance
    }
end

local function isVisible(from, part)
    local rayParams = RaycastParams.new()
    rayParams.FilterType = Enum.RaycastFilterType.Blacklist
    rayParams.FilterDescendantsInstances = {plr.Character}
    local result = workspace:Raycast(from, (part.Position - from).Unit * 500, rayParams)
    return result and result.Instance and result.Instance:IsDescendantOf(part.Parent)
end

local function getClosestTarget()
    local closest, shortest = nil, FOV_RADIUS
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= plr and p.Character and p.Character:FindFirstChild("Head") then
            local head = p.Character.Head
            local pos, onScreen = cam:WorldToViewportPoint(head.Position)
            if onScreen then
                local dist = (Vector2.new(pos.X, pos.Y) - fovCircle.Position).Magnitude
                if dist < shortest then
                    shortest, closest = dist, p
                end
            end
        end
    end
    return closest
end

Players.PlayerRemoving:Connect(function(player)
    if espObjects[player] then
        for _, obj in pairs(espObjects[player]) do
            obj:Remove()
        end
        espObjects[player] = nil
    end
end)

RunService.RenderStepped:Connect(function()
    -- FOV Circle com cor vermelha escura fixa
    fovCircle.Color = ESP_COLOR
    fovCircle.Position = Vector2.new(cam.ViewportSize.X / 2, cam.ViewportSize.Y / 2)
    fovCircle.Radius = FOV_RADIUS

    -- ESP System com cor vermelha escura fixa
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= plr and player.Character and player.Character:FindFirstChild("HumanoidRootPart") and player.Character:FindFirstChild("Head") then
            createESP(player)
            local esp = espObjects[player]
            local char = player.Character
            local hrp, head, humanoid = char.HumanoidRootPart, char.Head, char:FindFirstChild("Humanoid")
            local tl = cam:WorldToViewportPoint(hrp.Position + Vector3.new(-2, 3, 0))
            local tr = cam:WorldToViewportPoint(hrp.Position + Vector3.new(2, 3, 0))
            local bl = cam:WorldToViewportPoint(hrp.Position + Vector3.new(-2, -3, 0))
            local br = cam:WorldToViewportPoint(hrp.Position + Vector3.new(2, -3, 0))
            local center = cam:WorldToViewportPoint(hrp.Position)

            if espEnabled and tl.Z > 0 and tr.Z > 0 and bl.Z > 0 and br.Z > 0 then
                if showBox then
                    esp.Box.Visible = true
                    esp.Box.PointA = Vector2.new(tl.X, tl.Y)
                    esp.Box.PointB = Vector2.new(tr.X, tr.Y)
                    esp.Box.PointC = Vector2.new(br.X, br.Y)
                    esp.Box.PointD = Vector2.new(bl.X, bl.Y)
                    esp.Box.Color = ESP_COLOR
                else
                    esp.Box.Visible = false
                end

                if showLine then
                    esp.Line.Visible = true
                    esp.Line.From = Vector2.new(cam.ViewportSize.X / 2, cam.ViewportSize.Y)
                    esp.Line.To = Vector2.new(center.X, center.Y)
                    esp.Line.Color = ESP_COLOR
                else
                    esp.Line.Visible = false
                end

                if showName then
                    esp.Name.Visible = true
                    esp.Name.Position = Vector2.new(center.X, center.Y - 60)
                    esp.Name.Text = player.Name
                    esp.Name.Color = ESP_COLOR
                else
                    esp.Name.Visible = false
                end

                if showHealth then
                    esp.Health.Visible = true
                    esp.Health.Position = Vector2.new(center.X, center.Y - 40)
                    esp.Health.Text = "HP " .. math.floor(humanoid.Health)
                    esp.Health.Color = ESP_COLOR
                else
                    esp.Health.Visible = false
                end

                if showDistance then
                    local dist = math.floor((hrp.Position - cam.CFrame.Position).Magnitude)
                    esp.Distance.Visible = true
                    esp.Distance.Position = Vector2.new(center.X, center.Y - 20)
                    esp.Distance.Text = "Distância " .. dist .. "m"
                    esp.Distance.Color = ESP_COLOR
                else
                    esp.Distance.Visible = false
                end
            else
                for _, obj in pairs(esp) do
                    obj.Visible = false
                end
            end
        end
    end

    if not enabled and not rageMode then
        targetText.Visible = false
        statusText.Visible = false
        return
    end
    
    if not plr.Character or not plr.Character:FindFirstChild("HumanoidRootPart") or not plr.Character:FindFirstChild("Humanoid") then
        targetText.Visible = false
        statusText.Visible = true
        statusText.Text = "Aguardando spawn..."
        statusText.Color = Color3.fromRGB(255, 255, 85)
        return
    end
    
    local humanoid = plr.Character:FindFirstChild("Humanoid")
    if not humanoid or humanoid.Health <= 0 then
        targetText.Visible = false
        statusText.Visible = true
        statusText.Text = "Morto - Aimbot desativado"
        statusText.Color = Color3.fromRGB(255, 85, 85)
        return
    end
    
    local weapon = plr.Character:FindFirstChildWhichIsA("Tool")
    if not weapon then
        targetText.Visible = false
        statusText.Visible = true
        statusText.Text = "Sem Arma - Equipe uma arma"
        statusText.Color = Color3.fromRGB(255, 170, 85)
        return
    end
    
    statusText.Visible = false
    
    if tick() - lastShotTime < SHOT_DELAY then return end

    local target = getClosestTarget()
    if target and target.Character then
        local function getTargetPart(character)
            local head = character:FindFirstChild("Head")
            local torso = character:FindFirstChild("UpperTorso") or character:FindFirstChild("Torso")

            if not head or not torso then return nil end

            -- RAGE MODE: sempre cabeça (100%)
            if rageMode then
                return head
            end
            
            -- LEGIT MODE: 25% cabeça, 75% peito
            local chance = math.random()
            if chance <= 0.25 then
                return head
            else
                return torso
            end
        end

        local bodyPart = getTargetPart(target.Character)
        local humanoid = target.Character:FindFirstChild("Humanoid")

        if bodyPart and humanoid and isVisible(plr.Character.Head.Position, bodyPart) then
            targetText.Visible = true
            targetText.Text = "Alvo: " .. target.Name .. (rageMode and " [RAGE]" or " [LEGIT]")

            local args = {
                [1] = weapon,
                [2] = {
                    ["p"] = bodyPart.Position,
                    ["pid"] = 1,
                    ["part"] = bodyPart,
                    ["d"] = (plr.Character.Head.Position - bodyPart.Position).Magnitude,
                    ["maxDist"] = 999,
                    ["h"] = humanoid,
                    ["m"] = Enum.Material.Plastic,
                    ["sid"] = 5,
                    ["t"] = tick() % 1,
                    ["n"] = (bodyPart.Position - plr.Character.Head.Position).Unit
                }
            }

            if ReplicatedStorage:FindFirstChild("WeaponsSystem") then
                ReplicatedStorage.WeaponsSystem.Network.W__it:FireServer(unpack(args))
            end
            lastShotTime = tick()
        else
            targetText.Visible = false
        end
    else
        targetText.Visible = false
    end
end)

    function Window:CreateSlider(options)
        local Slider = Instance.new("Frame")
        Slider.Name = "Slider"
        Slider.Parent = ScrollingFrame
        Slider.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
        Slider.BorderSizePixel = 0
        Slider.Size = UDim2.new(1, 0, 0, 50)
        
        local SliderCorner = Instance.new("UICorner")
        SliderCorner.CornerRadius = UDim.new(0, 8)
        SliderCorner.Parent = Slider
        
        local SliderLabel = Instance.new("TextLabel")
        SliderLabel.Name = "SliderLabel"
        SliderLabel.Parent = Slider
        SliderLabel.BackgroundTransparency = 1
        SliderLabel.Position = UDim2.new(0, 10, 0, 0)
        SliderLabel.Size = UDim2.new(1, -20, 0, 20)
        SliderLabel.Font = Enum.Font.GothamBold
        SliderLabel.Text = options.Name or "Slider"
        SliderLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        SliderLabel.TextSize = 13
        SliderLabel.TextXAlignment = Enum.TextXAlignment.Left
        
        local SliderValue = Instance.new("TextLabel")
        SliderValue.Name = "SliderValue"
        SliderValue.Parent = Slider
        SliderValue.BackgroundTransparency = 1
        SliderValue.Position = UDim2.new(0, 10, 0, 0)
        SliderValue.Size = UDim2.new(1, -20, 0, 20)
        SliderValue.Font = Enum.Font.GothamBold
        SliderValue.Text = tostring(options.CurrentValue or 0) .. (options.Suffix or "")
        SliderValue.TextColor3 = Color3.fromRGB(220, 20, 20)
        SliderValue.TextSize = 13
        SliderValue.TextXAlignment = Enum.TextXAlignment.Right
        
        local SliderTrack = Instance.new("Frame")
        SliderTrack.Name = "SliderTrack"
        SliderTrack.Parent = Slider
        SliderTrack.BackgroundColor3 = Color3.fromRGB(10, 10, 10)
        SliderTrack.BorderSizePixel = 0
        SliderTrack.Position = UDim2.new(0, 10, 0, 25)
        SliderTrack.Size = UDim2.new(1, -20, 0, 15)
        
        local TrackCorner = Instance.new("UICorner")
        TrackCorner.CornerRadius = UDim.new(0, 8)
        TrackCorner.Parent = SliderTrack
        
        local SliderFill = Instance.new("Frame")
        SliderFill.Name = "SliderFill"
        SliderFill.Parent = SliderTrack
        SliderFill.BackgroundColor3 = Color3.fromRGB(220, 20, 20)
        SliderFill.BorderSizePixel = 0
        SliderFill.Size = UDim2.new(0, 0, 1, 0)
        
        local FillCorner = Instance.new("UICorner")
        FillCorner.CornerRadius = UDim.new(0, 8)
        FillCorner.Parent = SliderFill
        
        local currentValue = options.CurrentValue or options.Range[1]
        local minValue = options.Range[1]
        local maxValue = options.Range[2]
        local increment = options.Increment or 1
        
        local function updateSlider(value)
            value = math.clamp(value, minValue, maxValue)
            value = math.floor(value / increment) * increment
            currentValue = value
            
            local percentage = (value - minValue) / (maxValue - minValue)
            SliderFill.Size = UDim2.new(percentage, 0, 1, 0)
            SliderValue.Text = tostring(value) .. (options.Suffix or "")
            
            if options.Callback then
                options.Callback(value)
            end
        end
        
        updateSlider(currentValue)
        
        local dragging = false
        
        local function startSliderDrag(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging = true
                local percentage = math.clamp((input.Position.X - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X, 0, 1)
                local value = minValue + (maxValue - minValue) * percentage
                updateSlider(value)
            end
        end
        
        local function updateSliderDrag(input)
            if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                local percentage = math.clamp((input.Position.X - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X, 0, 1)
                local value = minValue + (maxValue - minValue) * percentage
                updateSlider(value)
            end
        end
        
        local function endSliderDrag(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging = false
            end
        end
        
        SliderTrack.InputBegan:Connect(startSliderDrag)
        UserInputService.InputChanged:Connect(updateSliderDrag)
        UserInputService.InputEnded:Connect(endSliderDrag)
        
        return Slider
    end
